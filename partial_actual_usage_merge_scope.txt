  139: func rawCostBasis(usage domain.TokenUsage, price domain.RoutePrice, mode InputPricingMode, modalities InputModalities) (int64, error) {
  140: 	var total int64
  141: 	addToken := func(count, unitPrice int64, name string) error {

  143: 			return fmt.Errorf("%w: %s", ErrMissingPrice, name)
  144: 		}

  146: 		if err != nil {
  147: 			return err
  148: 		}
  149: 		total, err = checkedAdd(total, component)
  150: 		return err
  151: 	}
  152: 	if mode == "" {
  153: 		mode = InputPricingModeDetailed
  154: 	}
  155: 	switch mode {
  156: 	case InputPricingModeDetailed:
  157: 		if err := addToken(usage.InputTokens, price.InputPricePer1MTokensCents, "input tokens"); err != nil {
  158: 			return 0, err
  159: 		}
  160: 		if err := addToken(usage.CachedInputTokens, price.CachedInputPricePer1MTokensCents, "cached input tokens"); err != nil {
  161: 			return 0, err
  162: 		}
  163: 		if err := addToken(usage.ImageInputTokens, price.ImageInputPricePer1MTokensCents, "image input tokens"); err != nil {
  164: 			return 0, err
  165: 		}
  166: 		if err := addToken(usage.AudioInputTokens, price.AudioInputPricePer1MTokensCents, "audio input tokens"); err != nil {
  167: 			return 0, err
  168: 		}
  169: 		if err := addToken(usage.FileInputTokens, price.FileInputPricePer1MTokensCents, "file input tokens"); err != nil {
  170: 			return 0, err
  171: 		}

  173: 			return 0, err
  174: 		}
  175: 	case InputPricingModeAggregateMax:
  176: 		if usage.CachedInputTokens != 0 || usage.ImageInputTokens != 0 || usage.AudioInputTokens != 0 || usage.FileInputTokens != 0 || usage.VideoInputTokens != 0 {
  177: 			return 0, fmt.Errorf("%w: aggregate input includes breakdown tokens", ErrInvalidUsage)
  178: 		}
  179: 		inputPrice, err := aggregateInputPrice(usage.InputTokens, price, modalities)
  180: 		if err != nil {
  181: 			return 0, err

  184: 			return 0, err
  185: 		}
  186: 	default:
  187: 		return 0, fmt.Errorf("%w: input pricing mode", ErrInvalidPricingInput)

  190: 		return 0, err
  191: 	}

  193: 		return 0, err
  194: 	}
  195: 	if err := addToken(usage.AudioOutputTokens, price.AudioOutputPricePer1MTokensCents, "audio output tokens"); err != nil {
  196: 		return 0, err
  197: 	}
  198: 	if usage.ImageGenerationUnits > 0 {

  200: 			return 0, fmt.Errorf("%w: image generation units require generated_image unit price", ErrInvalidImageGenerationPricing)
  201: 		}

  203: 		if err != nil {
  204: 			return 0, err
  205: 		}
  206: 		component, err := checkedMul(unitsPrice, microCentsPerCent)
  207: 		if err != nil {
  208: 			return 0, err
  209: 		}
  210: 		total, err = checkedAdd(total, component)
  211: 		if err != nil {
  212: 			return 0, err
  213: 		}
  214: 	}
  215: 	return total, nil
  216: }

  291: func ceilRat(r *big.Rat) *big.Int {
  292: 	n := new(big.Int).Set(r.Num())
  293: 	d := new(big.Int).Set(r.Denom())
  294: 	q, rem := new(big.Int).QuoRem(n, d, new(big.Int))
  295: 	if rem.Sign() > 0 {
  296: 		q.Add(q, big.NewInt(1))
  297: 	}
  298: 	return q
  299: }
