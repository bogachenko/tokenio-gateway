  191: 		context.Background(),
  192: 		shutdownTimeout,
  193: 	)

  195: 	cancel()

  197: 		return nil
  198: 	}
  199: 
  200: 	closeErr := server.Close()
  201: 	return errors.Join(
  202: 		fmt.Errorf(
  203: 			"graceful HTTP shutdown: %w",
  204: 			shutdownErr,
  205: 		),
  206: 		wrapRuntimeError(
  207: 			"force HTTP close",
  208: 			closeErr,
  209: 		),
  210: 	)
  211: }

  214: 	if err == nil ||
  215: 		errors.Is(err, http.ErrServerClosed) {

  217: 	}
  218: 	return fmt.Errorf("serve HTTP: %w", err)
  219: }

  231: 	}
